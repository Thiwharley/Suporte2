select codigo, descricao, cod_ncm, cod_cest, st,elo from estoque where codigo in
(select codigo from itevendas where data_emissao between '01.01.2023' and '31.12.2023')
 and cod_cest is null and st = '500'

(select codigo from itevendas where nota ='')
 and cod_cest is null and st = '500'
